\chapter{Hypermaps}

\sources{hypermap.v}{Hypermap.lean}

\begin{definition}[Hypermap]
  \label{def:Hypermap}
  \lean{Hypermap}
  \leanok
  A finite type with three permutations $e, n, f$ satisfying the triangular
  identity $n \circ f \circ e = \mathrm{id}$.
\end{definition}

\begin{theorem}[Triangular cancellation]
  \label{thm:Hypermap-cancel3}
  \lean{Hypermap.edgeK, Hypermap.faceK, Hypermap.nodeK}
  \leanok
  \uses{def:Hypermap}
  $n(f(e\,x)) = x$, $e(n(f\,x)) = x$, $f(e(n\,x)) = x$.
\end{theorem}
\begin{proof}\leanok\end{proof}

\begin{definition}[Genus and planarity]
  \label{def:Planar}
  \lean{Hypermap.genus, Hypermap.Planar}
  \leanok
  \uses{def:Hypermap}
  $\mathrm{genus}(G) = \tfrac{1}{2}(\mathrm{eulerLhs}\,G - \mathrm{eulerRhs}\,G)$;
  $G$ is planar iff $\mathrm{genus}(G) = 0$.
\end{definition}

\chapter{Real plane}

\sources{realplane.v}{RealPlane.lean}

\begin{definition}[Simple map]
  \label{def:SimpleMap}
  \lean{FourColor.RealPlane.SimpleMap, FourColor.RealPlane.FiniteSimpleMap}
  \leanok
  A simple map is a partial equivalence relation on $\R^2$ whose classes
  (the regions) are open and connected; finite if there are finitely many
  regions.
\end{definition}

\begin{definition}[$n$-colorable]
  \label{def:ColorableWith}
  \lean{FourColor.RealPlane.ColorableWith}
  \leanok
  \uses{def:SimpleMap}
  $m$ is $n$-colorable iff its regions admit a coloring by one of $n$
  colors with adjacent regions distinguished.
\end{definition}
